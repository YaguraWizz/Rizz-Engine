RIZZ Engine/         # Корневая директория проекта
├── bin/             # Скомпилированные исполняемые файлы
│   ├── build/       # Генерируемые файлы сборки (папка создаётся автоматически)    
├── setup/           # Директория скриптов сборки
│   ├── setup.bat    # Файл сборки запускающий CMAKE
├── editor/          # Директория IDE игрового движка (любого проекта/ может быть игрой)
│   ├── main.cpp     # Основной файл приложения
├── engen/           # Директория игрового движка
│   ├── Core/        # Основные утилиты и подсистемы (например, логирование, система событий)
│   ├── Graphics/    # Рендеринг (OpenGL, Vulkan, DirectX и т.д.)
│   ├── Physics/     # Физический движок (коллизии, динамика)
│   ├── Audio/       # Аудиосистема
│   ├── Input/       # Управление вводом (клавиатура, мышь, контроллеры)
│   ├── Math/        # Математические утилиты (векторы, матрицы и т.д.)
│   └── Utils/       # Вспомогательные утилиты    
├── third_party/     # Сторонние библиотеки
│   ├── glm/         # Пример: библиотека для работы с математикой
│   ├── stb/         # Пример: библиотека загрузки изображений
│   └── ...          # Другие зависимости
├── assets/          # Ассеты для тестов (текстуры, модели, шейдеры)
│   ├── textures/    # Текстуры
│   ├── models/      # 3D-модели
│   └── shaders/     # Шейдеры
├── tests/           # Тесты для движка
│   ├── unit/        # Юнит-тесты
│   ├── integration/ # Интеграционные тесты
│   └── examples/    # Примеры использования движка (мини-игры)
├── doc/             # Документация
├── CMakeLists.txt   # Основной файл CMake
├── README.md        # Документация/описание проекта
└── LICENSE          # Лицензия проекта
